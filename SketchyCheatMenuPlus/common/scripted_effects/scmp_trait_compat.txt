# Physique
add_physique_trait_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    add_trait = brawny
  }
  else = {
    add_trait = robust
  }
}
remove_physique_trait_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    remove_trait = brawny
  }
  else = {
    remove_trait = robust
  }
}

# Seducer (sex-aware)
add_seducer_trait_effect = {
  if = {
    limit = { is_female = yes }
    if = {
      limit = { has_global_flag = cleanslate_active }
      add_trait = master_seductress
    }
    else = {
      add_trait = seductress
    }
  }
  else = {
    if = {
      limit = { has_global_flag = cleanslate_active }
      add_trait = master_seducer
    }
    else = {
      add_trait = seducer
    }
  }
}
remove_seducer_trait_effect = {
  if = {
    limit = { is_female = yes }
    if = {
      limit = { has_global_flag = cleanslate_active }
      remove_trait = master_seductress
    }
    else = {
      remove_trait = seductress
    }
  }
  else = {
    if = {
      limit = { has_global_flag = cleanslate_active }
      remove_trait = master_seducer
    }
    else = {
      remove_trait = seducer
    }
  }
}
