############################################################
##                Sketchy Cheat Menu Plus                 ##
##              Engine-Mutex Cluster Helpers              ##
############################################################
##                                                        ##
## Strip-all-cluster-members helpers called from each     ##
## cluster member's add_X effect. Mirrors the engine's    ##
## `opposites` mutex with explicit if/limit-wrapped       ##
## removes so tooltips render `Lose Y` for the trait the  ##
## character actually has.                                ##
##                                                        ##
## Self-strip in each helper is safe: every add_X decision##
## gates on `NOT = { trait = X }` in its potential, so    ##
## the self-line never renders in the visible tooltip.    ##
##                                                        ##
############################################################

clear_intellect_traits_effect = {
  if = {
    limit = { trait = genius }
    remove_trait = genius
  }
  if = {
    limit = { trait = quick }
    remove_trait = quick
  }
  if = {
    limit = { trait = slow }
    remove_trait = slow
  }
  if = {
    limit = { trait = imbecile }
    remove_trait = imbecile
  }
}
