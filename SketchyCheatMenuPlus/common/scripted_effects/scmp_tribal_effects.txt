## --- CleanSlate rename compat hooks --- ##

# Aztec tribal
remove_aztec_tribal_buildings_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    remove_building = tb_culture_nahua_1
    remove_building = tb_culture_nahua_2
    remove_building = tb_culture_nahua_3
    remove_building = tb_culture_nahua_4
  }
  else = {
    remove_building = tb_culture_nahuatl_1
    remove_building = tb_culture_nahuatl_2
    remove_building = tb_culture_nahuatl_3
    remove_building = tb_culture_nahuatl_4
  }
}

# Aztec tribal (stack-aware add; building IDs differ per stack)
add_aztec_tribal_buildings_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    if = {
      limit = { location = { culture = nahua } }
      add_building = tb_culture_nahua_1
      add_building = tb_culture_nahua_2
      add_building = tb_culture_nahua_3
      add_building = tb_culture_nahua_4
    }
  }
  else = {
    if = {
      limit = { location = { culture = nahuatl } }
      add_building = tb_culture_nahuatl_1
      add_building = tb_culture_nahuatl_2
      add_building = tb_culture_nahuatl_3
      add_building = tb_culture_nahuatl_4
    }
  }
}

# tribal: religion / terrain / pillar-flag conditional

  # Defensive Fortifications (Baltic / Finnish / Slavic / West African / Bon pagan)
  if = {
    limit = {
      FROM = {
        OR = {
          religion = baltic_pagan
          religion = baltic_pagan_reformed
          religion = finnish_pagan
          religion = finnish_pagan_reformed
          religion = slavic_pagan
          religion = slavic_pagan_reformed
          religion = west_african_pagan
          religion = west_african_pagan_reformed
          is_bon_religion_trigger = yes
        }
      }
    }
    add_building = tb_defensive_fortifications_1
    add_building = tb_defensive_fortifications_2
    add_building = tb_defensive_fortifications_3
    add_building = tb_defensive_fortifications_4
  }

  # Desert terrain + desert-eligible culture
  if = {
    limit = {
      location = {
        terrain = desert
        OR = {
          culture_group = east_african
          culture_group = west_african
          culture_group = arabic
          culture_group = iranian
        }
      }
    }
    add_building = tb_desert_stable_1
    add_building = tb_desert_stable_2
    add_building = tb_desert_stable_3
    add_building = tb_desert_stable_4

  }

  if = {
