# Religion features that block Status of Women's visibility.
# Vanilla also excludes matriarchal religions; CleanSlate doesn't.
has_status_of_women_religion_block_trigger = {
  trigger_if = {
    limit = { has_global_flag = cleanslate_active }
    OR = {
      has_religion_feature = religion_equal
      has_religion_feature = religion_feature_bon
    }
  }
  trigger_else = {
    OR = {
      has_religion_feature = religion_matriarchal
      has_religion_feature = religion_equal
      has_religion_feature = religion_feature_bon
    }
  }
}

# Religion or government that blocks Set Revoke Title: Religious.
# Vanilla also excludes cosmopolitan religions and Chinese Imperial;
# CleanSlate doesn't.
has_revoke_title_religious_block_trigger = {
  trigger_if = {
    limit = { has_global_flag = cleanslate_active }
    always = no
  }
  trigger_else = {
    OR = {
      has_religion_feature = religion_cosmopolitan
      government = chinese_imperial_government
    }
  }
}
