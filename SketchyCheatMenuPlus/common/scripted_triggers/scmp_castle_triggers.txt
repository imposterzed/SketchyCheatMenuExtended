## --- Culture-matching presence checks --- ##

# Returns yes when the holding has the tier-4 cultural building
# for its current culture (tier-4 implies 1-3 via vanilla's
# prerequisite chain). Cultures without a cultural-building
# family fall through to always = yes.

# castle: 34 cultural ladders
has_castle_culture_buildings_trigger = {
  trigger_if = {
    limit = { location = { culture = dutch } }
    has_building = ca_culture_dutch_4
  }

  trigger_else_if = {
    limit = { location = { culture = ethiopian } }
    has_building = ca_culture_ethiopian_4
  }

  trigger_else_if = {
    limit = { location = { is_frank_norman_german_ladder_culture_trigger = yes } }
    has_building = ca_culture_frank_norman_german_4
  }

  trigger_else_if = {
    limit = { location = { is_germanic_ladder_culture_trigger = yes } }
    has_building = ca_culture_germanic_4
  }

  trigger_else_if = {
    limit = { location = { is_altaic_ladder_culture_trigger = yes } }
    has_building = ca_culture_group_altaic_4
  }

  trigger_else_if = {
    limit = { location = { is_arabic_ladder_culture_trigger = yes } }
    has_building = ca_culture_group_arabic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = baltic
          culture_group = finno_ugric
        }
      }
    }
    has_building = ca_culture_group_baltic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        culture_group = byzantine
        NOT = { culture = coptic }
        NOT = { culture = arberian }
      }
    }
    has_building = ca_culture_group_byzantine_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = iberian } }
    has_building = ca_culture_group_iberian_4
  }

  trigger_else_if = {
    limit = { location = { is_north_germanic_ladder_culture_trigger = yes } }
    has_building = ca_culture_group_north_germanic_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = south_slavic } }
    has_building = ca_culture_group_south_slavic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = west_slavic
          culture = hungarian
        }
      }
    }
    has_building = ca_culture_hungarian_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = indo_aryan_group
          culture_group = dravidian_group
        }
      }
    }
    has_building = ca_culture_indian_4
  }

  trigger_else_if = {
    limit = { location = { culture = irish } }
    has_building = ca_culture_irish_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture = italian
          culture = dalmatian
        }
      }
    }
    has_building = ca_culture_italian_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = israelite } }
    has_building = ca_culture_jewish_4
  }

  trigger_else_if = {
    limit = { location = { is_moors_ladder_culture_trigger = yes } }
    has_building = ca_culture_moors_4
  }

  trigger_else_if = {
    limit = { location = { culture = nubian } }
    has_building = ca_culture_nubian_4
  }

  trigger_else_if = {
    limit = { location = { culture = pictish } }
    has_building = ca_culture_pictish_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = east_slavic } }
    has_building = ca_culture_russian_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture = english
          culture = welsh
        }
      }
    }
    has_building = ca_culture_saxon_english_4
  }

  trigger_else_if = {
    limit = { location = { culture = scottish } }
    has_building = ca_culture_scottish_4
  }

  trigger_else_if = {
    limit = { location = { culture = somali } }
    has_building = ca_culture_somali_4
  }

  trigger_else_if = {
    limit = { location = { culture = suebi } }
    has_building = ca_culture_suebi_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = west_african
          culture_group = central_african
          culture = daju
        }
      }
    }
    has_building = ca_culture_west_african_4
  }

  # Aztec (CleanSlate rename ï¿½ stack-aware presence check)
  trigger_else_if = {
    limit = {
      OR = {
        AND = { has_global_flag = cleanslate_active location = { culture = nahua } }
        AND = { NOT = { has_global_flag = cleanslate_active } location = { culture = nahuatl } }
      }
    }

  trigger_else_if = {
    limit = { location = { culture = jurchen } }
    has_building = ca_culture_altaic_jurchen_4
  }

  # Animal cultures (CleanSlate-only)
  trigger_else_if = {

    has_building = ca_culture_animal_4
  }

  trigger_else_if = {
    limit = { location = { culture = arberian } }
    has_building = ca_culture_arberian_4
  }

  trigger_else_if = {
    limit = { location = { culture = han } }
    has_building = ca_culture_chinese_han_4
  }

  trigger_else_if = {
    limit = { location = { culture = coptic } }
    has_building = ca_culture_coptic_4
  }

  trigger_else_if = {
    limit = { location = { culture = roman } }
    has_building = ca_culture_roman_4
  }

  trigger_else_if = {
    limit = { location = { culture = sardinian } }
    has_building = ca_culture_sardinian_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = tibetan_group } }
    has_building = ca_culture_tibetan_4
  }

  trigger_else = {
    always = yes
  }
}

## --- CleanSlate rename compat hooks --- ##

# Aztec castle top-tier presence (stack-aware; building IDs differ)
has_aztec_castle_culture_buildings_top_trigger = {
  trigger_if = {
    limit = { has_global_flag = cleanslate_active }
    has_building = ca_culture_nahua_4
  }
  trigger_else = {
    has_building = ca_culture_nahuatl_4
  }
}

## --- Special-building presence checks --- ##

# Returns yes when every applicable special-buildings family is
# at top tier.

# castle: bloodline / terrain / government / title conditional
has_castle_special_buildings_trigger = {
  # Theodosian Walls (Constantinople-only)
  trigger_if = {
    limit = { title = b_constantinople }
    has_building = ca_con_wall_6
  }

  # Oppressive Fortifications (HF blood-building bloodline)
  trigger_if = {
    limit = { FROM = { any_owned_bloodline = { has_bloodline_flag = bloodline_blood_building } } }
    has_building = ca_oppressive_fortifications_4
  }

  # Special Fortifications (HF castellan bloodline)
  trigger_if = {
    limit = { FROM = { any_owned_bloodline = { has_bloodline_flag = bloodline_castellan } } }
    has_building = ca_special_fortifications_4
  }

  # Steppe terrain + steppe-eligible culture
  trigger_if = {
    limit = {
      location = {
        terrain = steppe
        OR = {
          culture_group = altaic
          culture_group = magyar
          culture = alan
          culture_group = iranian
        }
      }
    }
    has_building = ca_steppe_barracks_6
    has_building = ca_steppe_stable_6
  }

  # Desert terrain + desert-eligible culture
  trigger_if = {
    limit = {
      location = {
        terrain = desert
        OR = {
          culture_group = east_african
          culture_group = west_african
          culture_group = arabic
        }
      }
    }
    has_building = ca_desert_stable_6
  }

  # Monastic-feudal government
  trigger_if = {
    limit = { FROM = { is_monastic_feudal_government_trigger = yes } }
    has_building = ca_government_monastery_4
  }
}
