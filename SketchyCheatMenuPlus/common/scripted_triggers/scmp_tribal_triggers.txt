## --- Culture-matching presence checks --- ##

# Returns yes when the holding has the tier-4 cultural building
# for its current culture (tier-4 implies 1-3 via vanilla's
# prerequisite chain). Cultures without a cultural-building
# family fall through to always = yes.

# tribal: 33 cultural ladders
has_tribal_culture_buildings_trigger = {
  trigger_if = {
    limit = { location = { culture = dutch } }
    has_building = tb_culture_dutch_4
  }

  trigger_else_if = {
    limit = { location = { culture = ethiopian } }
    has_building = tb_culture_ethiopian_4
  }

  trigger_else_if = {
    limit = { location = { is_frank_norman_german_ladder_culture_trigger = yes } }
    has_building = tb_culture_frank_norman_german_4
  }

  trigger_else_if = {
    limit = { location = { is_germanic_ladder_culture_trigger = yes } }
    has_building = tb_culture_germanic_4
  }

  trigger_else_if = {
    limit = { location = { is_altaic_ladder_culture_trigger = yes } }
    has_building = tb_culture_group_altaic_4
  }

  trigger_else_if = {
    limit = { location = { is_arabic_ladder_culture_trigger = yes } }
    has_building = tb_culture_group_arabic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = baltic
          culture_group = finno_ugric
        }
      }
    }
    has_building = tb_culture_group_baltic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        culture_group = byzantine
        NOT = { culture = coptic }
        NOT = { culture = arberian }
      }
    }
    has_building = tb_culture_group_byzantine_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = iberian } }
    has_building = tb_culture_group_iberian_4
  }

  trigger_else_if = {
    limit = { location = { is_north_germanic_ladder_culture_trigger = yes } }
    has_building = tb_culture_group_north_germanic_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = south_slavic } }
    has_building = tb_culture_group_south_slavic_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = west_slavic
          culture = hungarian
        }
      }
    }
    has_building = tb_culture_hungarian_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = indo_aryan_group
          culture_group = dravidian_group
        }
      }
    }
    has_building = tb_culture_indian_4
  }

  trigger_else_if = {
    limit = { location = { culture = irish } }
    has_building = tb_culture_irish_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture = italian
          culture = dalmatian
        }
      }
    }
    has_building = tb_culture_italian_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = israelite } }
    has_building = tb_culture_jewish_4
  }

  trigger_else_if = {
    limit = { location = { is_moors_ladder_culture_trigger = yes } }
    has_building = tb_culture_moors_4
  }

  trigger_else_if = {
    limit = { location = { culture = nubian } }
    has_building = tb_culture_nubian_4
  }

  trigger_else_if = {
    limit = { location = { culture = pictish } }
    has_building = tb_culture_pictish_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = east_slavic } }
    has_building = tb_culture_russian_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture = english
          culture = welsh
        }
      }
    }
    has_building = tb_culture_saxon_english_4
  }

  trigger_else_if = {
    limit = { location = { culture = scottish } }
    has_building = tb_culture_scottish_4
  }

  trigger_else_if = {
    limit = { location = { culture = somali } }
    has_building = tb_culture_somali_4
  }

  trigger_else_if = {
    limit = { location = { culture = suebi } }
    has_building = tb_culture_suebi_4
  }

  trigger_else_if = {
    limit = {
      location = {
        OR = {
          culture_group = west_african
          culture_group = central_african
          culture = daju
        }
      }
    }
    has_building = tb_culture_west_african_4
  }

  # Aztec (CleanSlate rename â€” stack-aware presence check)
  trigger_else_if = {
    limit = {
      OR = {
        AND = { has_global_flag = cleanslate_active location = { culture = nahua } }
        AND = { NOT = { has_global_flag = cleanslate_active } location = { culture = nahuatl } }
      }
    }
    has_aztec_tribal_culture_buildings_top_trigger = yes
  }

  trigger_else_if = {
    limit = { location = { culture = jurchen } }
    has_building = tb_culture_altaic_jurchen_4
  }

  # Animal cultures (CleanSlate-only)
  trigger_else_if = {
    limit = { location = { is_animal_ladder_culture_trigger = yes } }
    has_building = tb_culture_animal_4
  }

  trigger_else_if = {
    limit = { location = { culture = arberian } }
    has_building = tb_culture_arberian_4
  }

  trigger_else_if = {
    limit = { location = { culture = han } }
    has_building = tb_culture_chinese_han_4
  }

  trigger_else_if = {
    limit = { location = { culture = coptic } }
    has_building = tb_culture_coptic_4
  }

  trigger_else_if = {
    limit = { location = { culture = sardinian } }
    has_building = tb_culture_sardinian_4
  }

  trigger_else_if = {
    limit = { location = { culture_group = tibetan_group } }
    has_building = tb_culture_tibetan_4
  }

  trigger_else = {
    always = yes
  }
}

## --- CleanSlate rename compat hooks --- ##

# Aztec tribal top-tier presence (stack-aware; building IDs differ)
has_aztec_tribal_culture_buildings_top_trigger = {
  trigger_if = {
    limit = { has_global_flag = cleanslate_active }
    has_building = tb_culture_nahua_4
  }
  trigger_else = {
    has_building = tb_culture_nahuatl_4
  }
}

## --- Special-building presence checks --- ##

# Returns yes when every applicable special-buildings family is
# at top tier.

# tribal: religion / terrain conditional
has_tribal_special_buildings_trigger = {
  # Defensive Fortifications (Baltic / Finnish / Slavic / West African / Bon pagan)
  trigger_if = {
    limit = {
      owner = {
        OR = {
          religion = baltic_pagan
          religion = baltic_pagan_reformed
          religion = finnish_pagan
          religion = finnish_pagan_reformed
          religion = slavic_pagan
          religion = slavic_pagan_reformed
          religion = west_african_pagan
          religion = west_african_pagan_reformed
          is_bon_religion_trigger = yes
        }
      }
    }
    has_building = tb_defensive_fortifications_4
  }

  # Desert terrain + desert-eligible culture
  trigger_if = {
    limit = {
      location = {
        terrain = desert
        OR = {
          culture_group = east_african
          culture_group = west_african
          culture_group = arabic
          culture_group = iranian
        }
      }
    }
    has_building = tb_desert_stable_4
  }

  # Great Pillar (owner + liege religion + tribal)
  trigger_if = {
    limit = {
      owner = {
        is_aztec_unreformed_religion_trigger = yes
        is_tribal = yes
        liege = {
          is_aztec_unreformed_religion_trigger = yes
          is_tribal = yes
        }
      }
    }
    has_building = tb_great_pillar_aztec
  }

  trigger_if = {
    limit = {
      owner = {
        religion = baltic_pagan
        is_tribal = yes
        liege = {
          religion = baltic_pagan
          is_tribal = yes
        }
      }
    }
    has_building = tb_great_pillar_baltic

  trigger_if = {
    limit = {
      owner = {
        is_bon_unreformed_religion_trigger = yes

          is_bon_unreformed_religion_trigger = yes

          is_tribal = yes

    }
    has_building = tb_great_pillar_finnish
  }

  trigger_if = {
    limit = {
      owner = {
        religion = hellenic_pagan
        is_tribal = yes
        liege = {
          religion = hellenic_pagan
          is_tribal = yes
        }
      }
    }
    has_building = tb_great_pillar_hellenic
  }

  trigger_if = {
    limit = {

        religion = norse_pagan

        liege = {
          religion = norse_pagan
          is_tribal = yes
        }
      }
    }
    has_building = tb_great_pillar_norse
  }

  trigger_if = {
