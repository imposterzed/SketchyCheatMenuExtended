# Seducer (sex-aware)
has_seducer_trait_trigger = {
  trigger_if = {
    limit = { is_female = yes }
    trigger_if = {
      limit = { has_global_flag = cleanslate_active }
      trait = master_seductress
    }
    trigger_else = {
      trait = seductress
    }
  }
  trigger_else = {
    trigger_if = {
      limit = { has_global_flag = cleanslate_active }
      trait = master_seducer
    }
    trigger_else = {
      trait = seducer
    }
  }
}
