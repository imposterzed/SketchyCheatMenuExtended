############################################################
##                Sketchy Cheat Menu Plus                 ##
##             Cheats Menu - Title Decisions              ##
############################################################
##                                                        ##
## Right-click decisions on a title (filter = owned). Two ##
## title-grab decisions (add_claim, take_title) grant or  ##
## take the clicked title; 25 law-switch decisions set    ##
## succession + gender laws on it (14 succession + 1      ##
## vanilla-label variant + 5 gender + 5 gender-hidden).   ##
## All gated by has_character_flag = cheats_enabled       ##
## (master toggle in cheats_menu.txt).                    ##
##                                                        ##
############################################################

title_decisions = {
  add_claim = {
    filter = owned

    from_potential = {
      ai = no
      has_character_flag = cheats_enabled
    }
    potential = {
      is_titular = no
      temporary = no
      higher_real_tier_than = BARON
      NOT = { holder = FROM }
    }
    allow = {
      always = yes
    }
    effect = {
      add_claim = FROM
    }
    ai_will_do = {
      factor = 0
    }
  }

  # Vanilla's Inheritance UI hides gender on these governments; engine
  # still applies. custom_tooltip renders the in-effect message.
  ## ===== Gender succession (Hidden on Iqta / Holy Order / Merchant Republic) ===== ##

  make_agnatic_hidden = {
    filter = owned

    from_potential = {
      ai = no
      has_character_flag = cheats_enabled
    }
    potential = {
      has_holder = yes
      NOT = { has_law = agnatic_succession }
      is_titular = no
      temporary = no
      holder_scope = {
        OR = {
          is_iqta_government_trigger = yes
          government = order_government
          government = merchant_republic_government
        }
      }
    }
    allow = {
      always = yes
    }
    effect = {
      holder_scope = {
        if = {
          limit = { is_iqta_government_trigger = yes }
          custom_tooltip = { text = gender_hidden_iqta_tooltip }
        }
        else_if = {
          limit = { government = order_government }
          custom_tooltip = { text = gender_hidden_order_tooltip }
        }
        else = {
          custom_tooltip = { text = gender_hidden_mr_tooltip }
        }
      }
      hidden_tooltip = { add_law = agnatic_succession }
    }
    ai_will_do = {
      factor = 0
    }
  }
