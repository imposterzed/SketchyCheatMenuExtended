############################################################
##                Sketchy Cheat Menu Plus                 ##
##              Cheats Menu - Title Decisions             ##
############################################################
##                                                        ##
## Right-click decisions on a title (filter = owned). Two ##
## title-grab decisions (add_claim, take_title) and 13    ##
## law switches (8 succession + 5 gender-succession). All ##
## gated by has_character_flag = cheats_enabled (master   ##
## toggle in cheats_menu.txt).                            ##
##                                                        ##
## higher_real_tier_than = COUNT gates the law switches   ##
## to Duke+ because counties inherit the de jure liege's  ##
## laws and changing them is a no-op.                     ##
##                                                        ##
############################################################

title_decisions = {

    filter = owned

    from_potential = {
      ai = no
      has_character_flag = cheats_enabled
      NOT = { owns = ROOT }
    }
    potential = {
      is_titular = no
      temporary = no
      higher_real_tier_than = BARON
    }
    allow = {
      always = yes
    }
    effect = {
