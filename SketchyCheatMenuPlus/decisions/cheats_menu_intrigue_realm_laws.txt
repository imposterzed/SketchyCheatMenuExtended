##                Sketchy Cheat Menu Plus                 ##
##                Cheats Menu - Realm Laws                ##
############################################################
##                                                        ##
## Right-click on self decisions for setting realm laws.  ##
## show_realm_laws / hide_realm_laws is the local toggle. ##
## Master cheats_enabled toggle lives in cheats_menu.txt. ##
##                                                        ##

decisions = {

  show_realm_laws = {

    potential = {
      ai = no
      has_character_flag = cheats_enabled
      NOT = { has_character_flag = show_realm_laws }
      has_scmp_realm_law_eligible_government_trigger = yes
    }

    allow = {
      always = yes
    }

    effect = {
      set_character_flag = show_realm_laws
    }

    revoke_allowed = {
      always = no
    }

    ai_will_do = {
      factor = 0
    }
  }

  hide_realm_laws = {

    potential = {
      ai = no
      has_character_flag = cheats_enabled
      has_character_flag = show_realm_laws
    }

    allow = {
      always = yes
    }

    effect = {
      clr_character_flag = show_realm_laws
    }

    revoke_allowed = {
      always = no
    }

    ai_will_do = {
      factor = 0
    }

  ## --- Crown Authority --- ##

  cycle_crown_authority_up = {

    potential = {
      ai = no
      has_character_flag = cheats_enabled
      has_character_flag = show_realm_laws
      NOT = { has_dlc = "Conclave" }
      NOT = { government = tribal_government }
      NOT = { government = nomadic_government }
      independent = yes
      NOT = { primary_title = { has_law = crown_authority_4 } }
    }

    allow = {
      always = yes
    }

    effect = {
      primary_title = {
        if = {
          limit = { has_law = crown_authority_3 }
          add_law = crown_authority_4
        }
        else_if = {
          limit = { has_law = crown_authority_2 }
          add_law = crown_authority_3
        }
        else_if = {
          limit = { has_law = crown_authority_1 }
