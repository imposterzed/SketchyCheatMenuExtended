  # Strip source-type buildings before convert_to to avoid
  # the orphan-building freeze. Potential blocks only the
  # realm capital; `is_capital = yes` alone matches any
  # county primary holding, not just the realm capital.
  convert_castle = {
    filter = realm_owned

    from_potential = {
      ai = no
      has_character_flag = cheats_enabled
    }
    potential = {
      NOT = {
        AND = {
          is_capital = yes
          FROM = { capital_holding = { title = ROOT } }
        }
      }
      NOT = { holding_type = castle }
    }
    allow = {
      always = yes
    }
    effect = {
      if = { limit = { holding_type = temple } strip_temple_buildings_effect = yes }
      else_if = { limit = { holding_type = city } strip_city_buildings_effect = yes }
      else_if = { limit = { holding_type = tribal } strip_tribal_buildings_effect = yes }
      else_if = { limit = { holding_type = nomad } strip_nomad_buildings_effect = yes }
      convert_to = CASTLE
    }
    ai_will_do = {
      factor = 0
