############################################################
##                Sketchy Cheat Menu Plus                 ##
##             Cheats Menu - Inheritance Laws             ##
############################################################
##                                                        ##
## Right-click on self decisions for setting inheritance  ##
## laws. show_inheritance_laws / hide_inheritance_laws is ##
## the local visibility toggle. Master cheats_enabled     ##
## toggle lives in cheats_menu.txt.                       ##
##                                                        ##
## Empirical: engine accepts add_law for every succession ##
## and gender law on every gov in our master OR-list.     ##
## Vanilla potentials gate the Laws-tab UI display, not   ##
## engine add_law. So per-decision gates are minimal.     ##
## DLC gates kept only where law genuinely requires DLC.  ##
##                                                        ##
############################################################

decisions = {
  show_inheritance_laws = {
    potential = {
      ai = no
      has_character_flag = cheats_enabled
      NOT = { has_character_flag = show_inheritance_laws }
      has_scmp_inheritance_law_eligible_government_trigger = yes
    }

    allow = {
      always = yes
    }

    effect = {
      set_character_flag = show_inheritance_laws
    }

    revoke_allowed = {
      always = no
    }

    ai_will_do = {
      factor = 0
    }
  }

  hide_inheritance_laws = {
    potential = {
      ai = no
      has_character_flag = cheats_enabled
      has_character_flag = show_inheritance_laws
    }

    allow = {
      always = yes
    }

    effect = {
      clr_character_flag = show_inheritance_laws
    }

    revoke_allowed = {
      always = no
    }

    ai_will_do = {
      factor = 0
    }
  }
