############################################################
##              Sketchy Cheat Menu Extended               ##
##                      Debug Events                      ##
############################################################
##                                                        ##
## Dev/test events for hard-to-reach paths. Never wired   ##
## to a decision or on_action – fire from the in-game     ##
## console:                                               ##
##   event SCMED.<id> <charID>                            ##
##                                                        ##
## Each event is a one-shot, no event_target deps,        ##
## hide_window so it fires silently.                      ##
##                                                        ##
############################################################

namespace = SCMED


# SCMED.1 – Assign the target character to dynasty 777777
# ("Outcast"). Used to verify the dynasty rename displays
# correctly, since no in-mod decision actually assigns
# characters to dynasty 777777 (disown_them reverts to the
# other parent's dynasty).
#
# Console:  event SCMED.1 <charID>
character_event = {
  id = SCMED.1
  hide_window = yes
  is_triggered_only = yes

  immediate = {
    dynasty = 777777
  }
}

# SCMED.2 – Grant the HF blood-building bloodline (carries
# bloodline_blood_building flag, gate for ca_oppressive_fortifications).
# Tests v0.9.1 add_castle_special_buildings_effect's Oppressive branch.
# Stack-aware: CleanSlate drops the `_01` suffix from bloodline IDs.
#
# Console:  event SCMED.2 <charID>
character_event = {
  id = SCMED.2
  hide_window = yes
  is_triggered_only = yes

  immediate = {
    if = {
      limit = { has_global_flag = cleanslate_active }
      if = {
        limit = { is_female = yes }
        create_bloodline = {
          type = hybrid_murder_builder_bloodline
          inheritance = matrilineal
        }
      }
      else = {
        create_bloodline = { type = hybrid_murder_builder_bloodline }
      }
    }
    else = {
      if = {
        limit = { is_female = yes }
        create_bloodline = {
          type = hybrid_murder_builder_bloodline_01
          inheritance = matrilineal
        }
      }
      else = {
        create_bloodline = { type = hybrid_murder_builder_bloodline_01 }
      }
    }
  }
}

# SCMED.6 – Dedicate the temple in this province to Zeus.
# Sets flag_dedicated_to_zeus on the temple title; gate for
# tp_shrine_zeus in v0.9.1's Hellenic branch of
# add_temple_special_buildings_effect.
#
# Console:  event SCMED.6 <provinceID>
province_event = {
  id = SCMED.6
  hide_window = yes
  is_triggered_only = yes

  immediate = {
    random_province_holding = {
      limit = { holding_type = temple }
      save_event_target_as = scoped_temple_barony
    }
    event_target:scoped_temple_barony = {
      set_title_flag = flag_dedicated_to_zeus
    }
  }
}
