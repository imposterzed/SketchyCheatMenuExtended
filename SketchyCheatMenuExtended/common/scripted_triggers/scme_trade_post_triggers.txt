## --- Special-building presence checks --- ##

# Returns yes when every applicable trade-post-buildings
# family is at top tier.

# trade post: trade-route / port / culture conditional
# Called from province scope (location wrap in upgrade_trade_post_fully.potential).
has_trade_post_buildings_trigger = {
  # Port + Patrician (Trade Post / Enclave / Garrison)
  trigger_if = {
    limit = {
      port = yes
      trade_post_owner = { is_patrician = yes }
    }
    trade_post_has_building = tr_port_3
    trade_post_has_building = tr_enclave_3
    trade_post_has_building = tr_garrison_3
  }

  # Silk Road (Trading Post / Trade Offices / Mercenary Garrison)
  trigger_if = {
    limit = {
      trade_route = silk_road
      silk_road_trade_post_trigger = yes
    }
    trade_post_has_building = tr_silk_road_3
    trade_post_has_building = trade_offices_3
    trade_post_has_building = mercenary_garrison_3
  }

  # Silk Road + Radhanite (Radhanite Quarter)
  trigger_if = {
    limit = {
      trade_route = silk_road
      silk_road_trade_post_trigger = yes
      trade_post_owner = {
        OR = {
          culture_group = israelite
          religion_group = jewish_group
          culture = khazar
        }
      }
    }
    trade_post_has_building = radhanite_quarter_3
  }

  # Trans-Saharan market (Trading Post)
  trigger_if = {
    limit = {
      trade_route = trans_saharan_trade_route
      trans_saharan_trade_post_market_trigger = yes
    }
    trade_post_has_building = tr_sahara_3
  }

  # Trans-Saharan gold mine (Gold Mine)
  trigger_if = {
    limit = {
      trade_route = trans_saharan_trade_route
      trans_saharan_trade_post_gold_mine_trigger = yes
    }
    trade_post_has_building = tr_gold_mine_3
  }

  # Audaghost / Taghaza (Salt Mine)
  trigger_if = {
    limit = {
      trade_route = trans_saharan_trade_route
      OR = {
        province = c_aoudaghost
        province = c_taghaza
      }
    }
    trade_post_has_building = tr_salt_mine_1
  }

  # Timbuktu (Cloth Production)
  trigger_if = {
    limit = {
      trade_route = trans_saharan_trade_route
      province = c_timbuktu
    }
    trade_post_has_building = tr_cloth_production_1
  }
}
