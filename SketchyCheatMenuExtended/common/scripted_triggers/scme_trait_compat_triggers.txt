# Seducer (sex-aware)
has_seducer_trait_trigger = {
  trigger_if = {
    limit = { is_female = yes }
    trigger_if = {
      limit = { has_global_flag = cleanslate_active }
      trait = master_seductress
    }
    trigger_else = {
      trait = seductress
    }
  }
  trigger_else = {
    trigger_if = {
      limit = { has_global_flag = cleanslate_active }
      trait = master_seducer
    }
    trigger_else = {
      trait = seducer
    }
  }
}

# Aztec religion
is_aztec_religion_trigger = {
  trigger_if = {
    limit = { has_global_flag = cleanslate_active }
    OR = {
      religion = aztec_pagan
      religion = aztec_pagan_reformed
    }
  }
  trigger_else = {
    OR = {
      religion = aztec
      religion = aztec_reformed
    }
  }
}
