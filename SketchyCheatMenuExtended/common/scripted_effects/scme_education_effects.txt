############################################################
##              Sketchy Cheat Menu Extended               ##
##                  Education Effects                     ##
############################################################
##                                                        ##
## Clear helper for the 20 vanilla education traits       ##
## (4 tiers x 5 stats). Called by each set_X_education    ##
## decision to strip the player's current education       ##
## before adding the target tier-4 trait. Each remove is  ##
## wrapped in if/limit so tooltips render only the trait  ##
## actually held. martial_cleric (tier-2 learning) is     ##
## CleanSlate-renamed; routed via the existing            ##
## remove_martial_cleric_trait_effect compat helper.      ##
##                                                        ##
############################################################

clear_education_traits_effect = {
  if = { limit = { trait = amateurish_plotter } remove_trait = amateurish_plotter }
  if = { limit = { trait = flamboyant_schemer } remove_trait = flamboyant_schemer }
  if = { limit = { trait = intricate_webweaver } remove_trait = intricate_webweaver }
  if = { limit = { trait = elusive_shadow } remove_trait = elusive_shadow }

  if = { limit = { trait = naive_appeaser } remove_trait = naive_appeaser }
  if = { limit = { trait = underhanded_rogue } remove_trait = underhanded_rogue }
  if = { limit = { trait = charismatic_negotiator } remove_trait = charismatic_negotiator }
  if = { limit = { trait = grey_eminence } remove_trait = grey_eminence }

  if = { limit = { trait = indulgent_wastrel } remove_trait = indulgent_wastrel }
  if = { limit = { trait = thrifty_clerk } remove_trait = thrifty_clerk }
  if = { limit = { trait = fortune_builder } remove_trait = fortune_builder }
  if = { limit = { trait = midas_touched } remove_trait = midas_touched }

  if = { limit = { trait = misguided_warrior } remove_trait = misguided_warrior }
  if = { limit = { trait = tough_soldier } remove_trait = tough_soldier }
  if = { limit = { trait = skilled_tactician } remove_trait = skilled_tactician }
  if = { limit = { trait = brilliant_strategist } remove_trait = brilliant_strategist }

  if = { limit = { trait = detached_priest } remove_trait = detached_priest }
  remove_martial_cleric_trait_effect = yes
  if = { limit = { trait = scholarly_theologian } remove_trait = scholarly_theologian }
  if = { limit = { trait = mastermind_theologian } remove_trait = mastermind_theologian }
}
