# Physique
add_physique_trait_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    add_trait = brawny
  }
  else = {
    add_trait = robust
  }
}
remove_physique_trait_effect = {
  if = {
    limit = { has_global_flag = cleanslate_active }
    if = {
      limit = { trait = brawny }
      remove_trait = brawny
    }
  }
  else = {
    if = {
      limit = { trait = robust }
      remove_trait = robust
    }
  }
}
